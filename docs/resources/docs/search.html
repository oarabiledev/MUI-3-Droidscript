Material3 SearchBars

Use SearchBars to find information, they do not act like an appBar and should not be a replacement for one.

There are 4 types of searchbars, you will not have to use different calls, its taken care by the plugin.

  • Normal SearchBar With Left And Right Icon
  • SearchBar With Leading Icon & Avatar Icon
  • SearchBar With Leading Icon & 2 Rights Icons
  • SearchBar With Leading Icons & Avatar + Icon

To call a searchBar use

ui.addSearchBar(leadingIcon,trailingIcons, hint, width, parentLay)

Check The Example Code And Imitate

Examples Of SearchBar Types:: 


app.LoadPlugin('Material3')

function OnStart()
{
    const __materialDebug = app.GetAppPath().endsWith('/Material3');
    const _materialPath = __materialDebug ? '' : app.GetPrivateFolder('Plugins') + '/material3/';
    app.CreateMaterial3()
    lay = ui.createLayout('Linear','FillXY',1,1)
    
    img = _materialPath + 'images/img.jpeg';
    
    search = ui.addSearchBar('menu','search', null, 0.9, lay)
    search.SetMargins(null,0.05)
    
    searchWithIcon = ui.addSearchBar('menu',img,null,0.9,lay)
    searchWithIcon.SetMargins(null,0.05)
    
    searchWithTwoIcons = ui.addSearchBar('menu','person,settings',null,0.9,lay)
    searchWithTwoIcons.SetMargins(null,0.05)
    
    searchWithIconAvatar = ui.addSearchBar('menu',`cast,${img}`,null,0.9,lay)
    searchWithIconAvatar.SetMargins(null,0.05)
    
    app.AddLayout(lay)
}


from native import app
app.LoadPlugin('Material3')

def OnStart():
    __materialDebug = app.GetAppPath().endsWith('/Material3');
    _materialPath = __materialDebug ? '' : app.GetPrivateFolder('Plugins') + '/material3/';
    
    app.CreateMaterial3()
    lay = ui.createLayout('Linear','FillXY',1,1)
    
    img = _materialPath + 'images/img.jpeg';
    
    search = ui.addSearchBar('menu','search', None, 0.9, lay)
    search.SetMargins(None,0.05)
    
    searchWithIcon = ui.addSearchBar('menu',img,None,0.9,lay)
    searchWithIcon.SetMargins(None,0.05)
    
    searchWithTwoIcons = ui.addSearchBar('menu','person,settings',None,0.9,lay)
    searchWithTwoIcons.SetMargins(None,0.05)
    
    searchWithIconAvatar = ui.addSearchBar('menu',f"cast,{img}",None,0.9,lay)
    searchWithIconAvatar.SetMargins(None,0.05)
    
    app.AddLayout(lay)
